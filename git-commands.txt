git				listet alle Git Befehle auf

git init			erstellt Git Repository für den aktuellen Ordner

git status			zeigt Status des aktuellen Ordners an (sind neue Dateien hinzugekommen, geändert worden, etc.)

git add filename.ext		fügt angegebene Datei zur Staging Area hinzu
git add *.ext			fügt alle Dateien mit angegebener Endung im Ordner zur Staging Areahinzu
git add .			fügt alle Dateien im Ordner zur Staging Areahinzu
git rm --cached filename.ext	enfernt angegebene Datei aus Staging Area

git commit -m "message"		commited Staging Area und fügt angegebene Nachricht an Commit an



git branch branchname		erstellt einen neuen Branch "branchname" (wechselt aber nicht zu diesem)
git checkout branchname		wechselt zum Branch "branchname"
git merge branchname		fügt den Branch "branchname" zum aktuellen Branch hinzu


git remote add origin <URL>	verbindet ein lokales Repository mit GitHub
git push -u origin master	lädt lokales Repository zu GitHub hoch
git push			lädt Änderungen in lokalem Repository zu GitHub hoch
git pull			lädt Repository von GitHub herunter
git clone <URL>			lädt Repository als neuen Ordner herunter und legt dort neues lokales Repository an
